- 코드 구현(C++)

```cpp
#include <iostream>
#include <algorithm>
using namespace std;
int A[5] = { 10,6,1,3,4 };
void print_arr() {
	cout << "\n";
	for (int i = 0; i < 5; i++) cout << A[i] << " ";
	cout << "\n";
}
void quick_sort(int left, int right) {
	if (left == right) return;
	swap(A[left], A[((right - left) / 2 + left)]);
	int l = left + 1;
	int r = right - 1;
	int pivot = left;
	while (l <= r) {
		while (l <= r && A[pivot] >= A[l])
			l++;
		while (l <= r && A[r] >= A[pivot])
			r--;
		if (l <= r)
			swap(A[l], A[r]);
	}
	swap(A[pivot], A[r]);
	quick_sort(left, r);
	quick_sort(r + 1, right);
}

int main(void) {
	print_arr();
	quick_sort(0, 5);
	print_arr();
	return 0;
}
```
